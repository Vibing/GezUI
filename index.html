style公用样式说明


    style.css中存在的样式是常用公用型样式。

    为了方便记忆，我们采用单词首字母 - 值的形式命名。

    例如：font-weight:bold 我们取key部分的每个单词首字母加"-"，然后跟上想要的属性值bold 即结果为：fw-bold

    下面为大家举出几种常用样式的分类：


    font-family：
    .ff-simsun
    .ff-yahei
    .ff-georgia

    /*字体大小*/
    font-size：
    .fs-12 至 .fs-30 以偶数2递增

    /*字体粗细*/
    font-weight：
    .fw-bold     //粗体
    .fw-normal   //正常

    /*字体颜色*/
    color：
    .c-000 至 c-fff   //纯黑-纯白

    /*背景颜色*/
    background-color:
    .bc-000 至 bc-fff //纯黑-纯白

    /*外边距、内边距*/
    margin和padding //10px以内 1-10都有 超出10px以外 以5px递增至50px

    margin-left:
    .m-*   //margin:xx
    .ml-*  //margin-left:xx

    /*行高*/
    .lh-2 至 .lh-60 //表示line-height:2 至 line-height:60 以2递增

    //除了以2递增 还有经常用的以5递增
    .lh-5 至 .lh-60 //表示line-height:5 至 line-height:60 以5递增

    /*宽度、高度*/
    width：
    .w-0 至 .w-100   //表示width:0px 至 width:100px 以5递增
    .w-100 至 .w-200 //表示width:100px 至 width:200px 以10递增
    高度类似


    除了上面提到的规律性命名 还有个别特殊命名：

    /*浮动*/
    float:left  -->  fl
    float:left  -->  fr

    /*字体浮动*/
    text-align:
    .tc //text-align:center
    .tr //text-align:right
    .tl //text-align:left

    /*清除浮动*/
    .clearfix

栅格系统

                                
    该框架在Bootstrap的栅格系统上做了修改 成为比较实用的css响应式布局框架
    栅格简介：http://v3.bootcss.com/css/#grid

    理解节点：
    超小屏幕（手机，小于 768px）
    没有任何媒体查询相关的代码，因为这在 Bootstrap 中是默认的（还记得 Bootstrap 是移动设备优先的吗？）

    小屏幕（平板，大于等于 768px）
    @media (min-width: @screen-m-min) { ... }

    中等屏幕（桌面显示器，大于等于 992px）
    @media (min-width: @screen-l-min) { ... }

    大屏幕（大桌面显示器，大于等于 1200px）
    @media (min-width: @screen-xl-min) { ... }

    我们偶尔也会在媒体查询代码中包含 max-width 从而将 CSS 的影响限制在更小范围的屏幕大小之内。
    @media (max-width: @screen-s-max) { ... }
    @media (min-width: @screen-m-min) and (max-width: @screen-s-max) { ... }
    @media (min-width: @screen-l-min) and (max-width: @screen-m-max) { ... }
    @media (min-width: @screen-xl-min) { ... }

    栅格分类介绍：

    为了记忆方便 将栅格系统的尺寸节点划分为衣服尺码 即：s、m、l、xl
    .col-s-* : 超小屏幕(手机,小于768px)
    .col-m-* : 小屏幕(平板,大于等于 768px)
    .col-l-* : 中等屏幕(桌面显示器,大于等于 992px)
    .col-xl-* : 大屏幕(大桌面显示器,大于等于 1200px)

    注意：row必须放在container或者container-fluid(100%)中
                                
                            
                                
    gez-ui.css的辅助类：针对不同屏幕尺寸隐藏或显示页面内容

    所有.visible-*-*默认为display:none;

    .visible-s-* : 屏幕小于等于767px时 显示
    .visible-m-* : 屏幕大于等于768px 并 小于等于 991px时 显示
    .visible-l-* ：屏幕大于等于992px 并 小于等于 1199时 显示
    .visible-xl-* : 屏幕大于等于1200px时 显示

    .hidden-s-* ： 屏幕小于等于767px时 隐藏

    .hidden-m-* ： 屏幕大于等于768px 并 小于等于991px时 隐藏
    .hidden-l-* ： 屏幕大于等于992px 并 小于等于1199px时 隐藏
    .hidden-xl-* ： 屏幕大于等于1200px时 隐藏

    同样以衣服尺码来记忆：s、m、l、xl

Switch开关

                                
    html: <input text-on="ON" text-off="OFF" type="checkbox" />
    启用：<input  type="checkbox" checked />
    禁用：<input  type="checkbox" disabled />

    js初始化调用：
    $('.switchbox').switch({
        beforeChange:function (obj, state, next) {
            console.log('before');
            next();//考虑到异步情况，这里采用nodejs中间件的方式，只有执行next()时，才会继续往下执行
        },
        afterChange: function(obj, state) {
            console.log('after');
        },
        change: function(obj, state) {
            console.log('change');
        }
    });

    $('.switchbox2').switch({
        beforeChange:function (obj, state, next) {
            console.log('before');
            //模拟ajax延迟
            setTimeout(next,1000);
        },
        afterChange: function(obj, state) {
            console.log('after');
        },
        change: function(obj, state) {
            console.log('change');
        }
    });

    你也可以对它的状态进行设置：
    $('.switchbox2').setSwitch({
            trigger: false, //状态改变时是否触发  beforeChange、afterChange、change回调
            checked: true,  //设置checked属性为true
            disabled: true  //设置disabled属性为true
    });

    $(".demoform").Validform({
        btnSubmit:"#btn_sub",
        btnReset:".btn_reset",
        tiptype:1,
        ignoreHidden:false,
        dragonfly:false,
        tipSweep:true,
        label:".label",
        showAllError:false,
        postonce:true,
        ajaxPost:true,
        datatype:{
            "*6-20": /^[^\s]{6,20}$/,
            "z2-4" : /^[\u4E00-\u9FA5\uf900-\ufa2d]{2,4}$/,
            "username":function(gets,obj,curform,regxp){
                /*
                    参数gets是获取到的表单元素值，obj为当前表单元素，curform为当前验证的表单，
                    regxp为内置的一些正则表达式的引用;
                */
                var reg1=/^[\w\.]{4,16}$/,
                reg2=/^[\u4E00-\u9FA5\uf900-\ufa2d]{2,8}$/;

                if(reg1.test(gets)){return true;}
                if(reg2.test(gets)){return true;}
                return false;
                /*
                注意return可以返回true 或 false 或 字符串文字，true表示验证通过，
                返回字符串表示验证失败，字符串作为错误提示显示，返回false则用errmsg或默认的错误提示;
                */
            },
            "phone":function(){
                // 5.0 版本之后，要实现二选一的验证效果，datatype 的名称 不 需要以 "option_" 开头;
            }
        },
        usePlugin:{
        swfupload:{},
        datepicker:{},
        passwordstrength:{},
        jqtransform:{
            selector:"select,input"
        }
        },
        beforeCheck:function(curform){
            //在表单提交执行验证之前执行的函数，curform参数是当前表单对象。
            //这里明确return false的话将不会继续执行验证操作;
        },
        beforeSubmit:function(curform){
            //在验证成功后，表单提交前执行的函数，curform参数是当前表单对象。
            //这里明确return false的话表单将不会提交;
        },
        callback:function(data){
            /*
                返回数据data是json对象，{"info":"demo info","status":"y"}
                info: 输出提示信息;
                status: 返回提交数据的状态,是否提交成功。
                如可以用"y"表示提交成功，"n"表示提交失败，在ajax_post.php文件返回数据里自定字符，
                主要用在callback函数里根据该值执行相应的回调操作;

                你也可以在ajax_post.php文件返回更多信息在这里获取，进行相应操作；
                ajax遇到服务端错误时也会执行回调，
                这时的data是{ status:**, statusText:**, readyState:**, responseText:** }；

                这里执行回调操作;
                注意：如果不是ajax方式提交表单，传入callback，这时data参数是当前表单对象，
                回调函数会在表单验证全部通过后执行，然后判断是否提交表单，
                如果callback里明确return false，则表单不会提交，如果return true或没有return，则会提交表单。
            */
        }
    });

图片上传预览


    //所有参数：
    $.Tool.uploadPreview({
        UpBtn:  //选择文件控件jquery对象;
        DivShow:    //DIV控件jquery对象;
        ImgShow:    //图片控件jquery对象;
        Width:  //预览宽度;
        Height: //预览高度;
        ImgType:    //支持文件类型 格式:["jpg","png"];
        callback:   //选择文件后回调方法;
    });

    //常用写法：
    $.Tool.uploadPreview({
        UpBtn:$('#up_img'),
        DivShow:$('#imgdiv'),
        ImgShow:$('#imgShow'),
        Width:500,
        Height:300,
        ImgType:["jpg","png"],
        callback:function(){
            var imgobj = new Image();

            alert('图片大小：'+ $.Tool.getImgSize($('#up_img'))+'k');
            imgobj.src = $('#imgShow').attr('src');

            imgobj.onload = function(){
                if(this.width<500 && this.height<300){
                    $.Tool.tips({type:2,content:"您上传的图片尺寸要大于500*300px"});
                    $('#imgShow').attr('src','');
                    $('#up_img').val('');
                }
            }
        }
    });

日期插件

                                
    使用方法可见：http://laydate.layui.com/

    默认调用：<input onclick="laydate()">
    时间：
                                
                            
                                
    规定格式：<input onclick="laydate({istime: true, format: 'YYYY-MM-DD hh:mm:ss'})">
    时间：
                                
                            
                                
    开始日：<input  id="start" style="width:200px; margin-right:10px;">
    结束日：<input id="end" style="width:200px;">
    var start = {
        elem: '#start',
        format: 'YYYY/MM/DD hh:mm:ss',
        min: laydate.now(), //设定最小日期为当前日期 或者：laydate.now(-1), -1代表昨天，-2代表前天，以此类推
        max: '2099-06-16 23:59:59', //最大日期
        istime: true,
        istoday: false,
        choose: function(datas){
            end.min = datas; //开始日选好后，重置结束日的最小日期
            end.start = datas //将结束日的初始值设定为开始日
        }
    };
    var end = {
        elem: '#end',
        format: 'YYYY/MM/DD hh:mm:ss',
        min: laydate.now(),
        max: '2099-06-16 23:59:59',
        istime: true,
        istoday: false,
        choose: function(datas){
            start.max = datas; //结束日选好后，重置开始日的最大日期
        }
    };
    laydate(start);
    laydate(end);
